	CREATE
	VIEW t2tierp.VIEW_COMPRA_REQ_ITEM_COTADO
	AS
	(
	select
		CRD.ID,
		CRD.ID_COMPRA_REQUISICAO,
		CRD.ID_PRODUTO,
		P.NOME AS 'PRODUTO.NOME',
		CRD.QUANTIDADE,
		CRD.QUANTIDADE_COTADA,
		CRD.ITEM_COTADO,
		COT.ID AS 'ID_COTACAO'
	from
		COMPRA_REQUISICAO_DETALHE CRD, COMPRA_REQ_COTACAO_DETALHE CRCD, PRODUTO P, COMPRA_COTACAO COT
	where
		CRD.ID = CRCD.ID_COMPRA_REQUISICAO_DETALHE and
		CRD.ID_PRODUTO = P.ID and
		COT.ID = CRCD.ID_COMPRA_COTACAO
	)
