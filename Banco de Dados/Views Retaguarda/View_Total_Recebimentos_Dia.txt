CREATE VIEW VIEW_FIN_TOTAL_RECEBIMENTOS_DIA(
	ID_CONTA_CAIXA,
    DATA_RECEBIMENTO,
    TOTAL)
AS
select
  PR.ID_CONTA_CAIXA, PR.DATA_RECEBIMENTO, SUM(PR.VALOR_RECEBIDO) AS TOTAL
from
  FIN_PARCELA_RECEBIMENTO PR
group by ID_CONTA_CAIXA, DATA_RECEBIMENTO;
