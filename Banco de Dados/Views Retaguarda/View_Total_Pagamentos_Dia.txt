CREATE VIEW VIEW_FIN_TOTAL_PAGAMENTOS_DIA(
	ID_CONTA_CAIXA,
    DATA_PAGAMENTO,
    TOTAL)
AS
select
  PP.ID_CONTA_CAIXA, PP.DATA_PAGAMENTO, SUM(PP.VALOR_PAGO) AS TOTAL
from
  FIN_PARCELA_PAGAMENTO PP
group by ID_CONTA_CAIXA, DATA_PAGAMENTO;
